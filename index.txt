    <script>
        const pilotos = { "101": { n: "Juan Perez" }, "102": { n: "Pedro Marmol" } };

        function entrar() {
            if(document.getElementById('u').value.toLowerCase().trim()=="op1" && document.getElementById('p').value=="terpac1") {
                document.getElementById('login').style.display='none'; 
                document.getElementById('ui-app').style.display='block';
            } else { alert("Acceso Incorrecto"); }
        }

        function check() {
            const id = document.getElementById('id-in').value.trim();
            const box = document.getElementById('piloto-box');
            if(pilotos[id]) { 
                box.innerText = "👨‍✈️ Piloto: " + pilotos[id].n; 
                box.style.display='block'; 
            } else { 
                box.style.display='none'; 
            }
        }

        async function enviar() {
            const id = document.getElementById('id-in').value.trim();
            if(!pilotos[id]) return alert("ID no encontrado");

            const btn = document.getElementById('btn-env');
            btn.innerText = "Enviando...";
            btn.disabled = true;

            const APP_ID = 'app3cdqCOInpldMCJ';
            const TABLE_ID = 'tblElQcwWk5BvQiX8';
            const TOKEN = 'pat4AVN7AOIYp9iCj.c6b58fa271e8e07e6291d08b2c6a1e0d54a49166ff80ef7a0fb73f1b4aefd55c';

            const infoTexto = `ID: ${id} | Piloto: ${pilotos[id].n} | Mov: ${document.getElementById('mov').value} | Prod: ${document.getElementById('prod').value}`;
            
            const cuerpoCarga = {
                "records": [{
                    "fields": {
                        "DATOS": infoTexto,
                        "FECHA": new Date().toISOString()
                    }
                }]
            };

            try {
                const response = await fetch(`https://api.airtable.com{APP_ID}/${TABLE_ID}`, {
                    method: 'POST',
                    headers: {
                        'Authorization': `Bearer ${TOKEN}`,
                        'Content-Type': 'application/json'
                    },
                    body: JSON.stringify(cuerpoCarga)
                });

                const resultado = await response.json();

                if (response.ok) {
                    alert("✅ REGISTRO ENVIADO A AIRTABLE");
                    location.reload();
                } else {
                    console.error(resultado);
                    alert("❌ ERROR DE AIRTABLE: " + (resultado.error.message || "Error de columnas"));
                    btn.disabled = false;
                    btn.innerText = "REINTENTAR";
                }
            } catch (error) {
                console.error(error);
                alert("❌ ERROR DE CONEXIÓN: Revisa tu internet.");
                btn.disabled = false;
                btn.innerText = "REINTENTAR";
            }
        }
    </script>
